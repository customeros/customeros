mutation MergeCustomFieldToFieldSet($contactId: ID!, $fieldSetId: ID!){
     customFieldMergeToFieldSet(
        contactId: $contactId,
        fieldSetId: $fieldSetId,
        input: {name: "some name"
                value: "some value"
                datatype: TEXT
                source: "manual"}
     ) {
        id
        name
        datatype
        value
        source
      }
}
