query GetTimelineEvents($ids: [ID!]!) {
  timelineEvents(ids: $ids) {
    ... on PageView {
      id
      application
      startedAt
      endedAt
      engagedTime
      pageUrl
      pageTitle
      orderInSession
      sessionId
      source
      sourceOfTruth
      appSource
    }
    ... on Issue {
      id
      subject
      status
      priority
      description
      source
      sourceOfTruth
      appSource
    }
    ... on Conversation {
      id
      startedAt
      subject
      channel
    }
    ... on Note {
      id
      html
      createdAt
    }
    ... on InteractionEvent {
      id
      content
      contentType
      channel
      createdAt
    }
    ... on Analysis {
      id
      content
      contentType
      analysisType
      createdAt
    }
  }
}
