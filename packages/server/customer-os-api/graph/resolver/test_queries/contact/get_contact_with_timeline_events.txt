query GetContactById($contactId: ID!, $from: Time!, $size: Int!){
  contact(id: $contactId) {
    id
    timelineEvents(from:$from, size:$size) {
            __typename
            ... on PageView {
              id
              application
              startedAt
              endedAt
              engagedTime
              pageUrl
              pageTitle
              orderInSession
              sessionId
            }
            __typename
            ... on Conversation {
                id
                startedAt
                subject
                channel
            }
            __typename
            ... on Note {
                id
                html
                createdAt
            }
            __typename
            ... on InteractionEvent {
                id
                content
                contentType
                channel
                createdAt
            }
            __typename
            ... on Analysis {
                id
                content
                contentType
                analysisType
                createdAt
            }
            __typename
            ... on Meeting {
                id
                name
                createdAt
            }
          }
  }
}
