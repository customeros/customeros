query GetContactById($contactId: ID!, $from: Time!, $to: Time!){
  contact(id: $contactId) {
    id
    actions(from:$from, to:$to) {
            __typename
            ... on PageViewAction {
              id
              application
              startedAt
              endedAt
              engagedTime
              pageUrl
              pageTitle
              orderInSession
              sessionId
            }
            __typename
            ... on Ticket {
              id
              createdAt
              updatedAt
              subject
              status
              priority
              description
              notes {
                id
                html
              }
              tags {
                id
                name
              }
            }
            __typename
            ... on Conversation {
                id
                startedAt
                subject
                channel
            }
          }
  }
}
