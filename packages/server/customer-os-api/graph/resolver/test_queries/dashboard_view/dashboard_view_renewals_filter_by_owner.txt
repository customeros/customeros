query dashboardView_Renewals($page: Int!, $limit: Int!, $ownerIdList: Any!, $ownerIdEmpty: Boolean!) {
  dashboardView_Renewals(
  pagination: { page: $page, limit: $limit },
  where: {
     AND: [
         {
             filter: {
                 property: "OWNER_ID",
                 operation: IN,
                 value: $ownerIdList
                 includeEmpty: $ownerIdEmpty
             }
         }
     ]
    },
    sort: {by: "OWNER"}) {
    content {
                      organization{
                         id
                } contract{
                id
                } opportunity{
                id
                }
                }
    totalAvailable
    totalElements
  }
}
