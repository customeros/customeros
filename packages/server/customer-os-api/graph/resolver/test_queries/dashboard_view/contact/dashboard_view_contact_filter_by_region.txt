query dashboardView_Contacts($page: Int!, $limit: Int!, $region1: Any!) {
  dashboardView_Contacts(
  pagination: { page: $page, limit: $limit },
  where: {
     AND: [
         {
             filter: {
                 property: "REGION",
                 operation: EQ,
                 value: $region1
             }
         }
     ]
    }) {
    content {
          id
          name
          firstName
          lastName
    }
    totalElements
  }
}
