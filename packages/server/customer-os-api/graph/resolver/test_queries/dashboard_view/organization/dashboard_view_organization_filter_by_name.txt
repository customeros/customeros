query dashboardView_Organizations($page: Int!, $limit: Int!, $searchTerm: Any!) {
  dashboardView_Organizations(
  pagination: { page: $page, limit: $limit },
  where: {
     AND: [
         {
             filter: {
                 property: "ORGANIZATION",
                 operation: EQ,
                 value: $searchTerm
             }
         }
     ]
    }) {
    content {
       id
        name
        description
        source
        industry
        locations {
            id
            name
            country
            region
            locality
        }
        website
        domains
        updatedAt
    }
    totalElements
  }
}
