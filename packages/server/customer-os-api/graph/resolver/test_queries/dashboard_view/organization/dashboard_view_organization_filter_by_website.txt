query dashboardView_Organizations($page: Int!, $limit: Int!, $searchTerm: Any!, $includeEmpty: Boolean!) {
  dashboardView_Organizations(
  pagination: { page: $page, limit: $limit },
  where: {
     AND: [
         {
             filter: {
                 property: "WEBSITE",
                 operation: EQ,
                 value: $searchTerm,
                 includeEmpty: $includeEmpty
             }
         }
     ]
    },
    sort: {by: "ORGANIZATION"}) {
    content {
       id
        name
        description
        source
        industry
        locations {
            id
            name
            country
            region
            locality
        }
        website
        domains
        updatedAt
    }
    totalElements
  }
}
