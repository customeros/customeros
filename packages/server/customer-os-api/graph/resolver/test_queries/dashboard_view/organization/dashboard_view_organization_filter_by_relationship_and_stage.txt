query dashboardView_Organizations(
  $page: Int!
  $limit: Int!
  $relationship: Any!
  $stage: Any!
) {
  dashboardView_Organizations(
    pagination: {
      page: $page
      limit: $limit
    }
    where: {
      AND: [
        {
          filter: {
            property: "RELATIONSHIP"
            operation: EQ
            value: $relationship
          }
        }
        {
          filter: {
            property: "STAGE"
            operation: EQ
            value: $stage
          }
        }
      ]
    }
    sort: {
      by: "RELATIONSHIP"
    }
  ) {
    content {
      id
      name
      description
      source
      industry
      locations {
        id
        name
        country
        region
        locality
      }
      website
      domains
      updatedAt
    }
    totalElements
  }
}
