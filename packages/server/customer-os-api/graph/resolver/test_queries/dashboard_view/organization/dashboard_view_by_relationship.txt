query RelationshipsOrganizations($relationship: Any!, $page: Int!, $limit: Int!) {
  dashboardView_Organizations(
    pagination: { page: $page, limit: $limit },
    sort: { by: "RELATIONSHIP" }
    where: {
         AND: [
             {
                 filter: {
                     property: "RELATIONSHIP",
                     operation: EQ,
                     value: $relationship
                 }
             }
         ]
        },
  ) {
    content {
      id
      name
    }
    totalElements
  }
}
