mutation JobRoleUpdate($contactId: ID!, $roleId: ID!){
     jobRole_Update(
        contactId: $contactId
        input: {
                    id: $roleId
                    jobTitle: "CEO"
                    description: "CEO of Apple"
                    company: "Apple"
                    primary: true
                    responsibilityLevel: 1
               }
     ) {
        organization {
            id
        }
        updatedAt
        jobTitle
        description
        company
        primary
        responsibilityLevel
        sourceOfTruth
        startedAt
        endedAt
      }
}
