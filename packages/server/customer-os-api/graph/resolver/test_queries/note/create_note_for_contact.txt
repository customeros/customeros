mutation CreateNote($contactId: ID!) {
  note_CreateForContact(
    contactId: $contactId,
    input: {
      content: "Note content"
      contentType: "text/markdown"
    }
  ) {
    id
    html
    content
    contentType
    createdAt
    updatedAt
    source
    sourceOfTruth
    appSource
    createdBy {
      id
    }
  }
}
