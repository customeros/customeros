mutation AddAttachmentToNote($noteId: ID!, $attachmentId: ID!) {
    note_LinkAttachment(
            noteId: $noteId,
            attachmentId: $attachmentId) {
      id
      content
      contentType
      updatedAt
      sourceOfTruth
      includes {
        id
        name
        size
        mimeType
        extension
      }
    }
}
