query GetOrganizationById($organizationId: ID!){
  organization(id: $organizationId) {
    id
    notes {
        content {
          id
          html
          content
          contentType
          mentioned {
            __typename
            ... on Issue {
                id
                subject
            }
          }
          source
          sourceOfTruth
          appSource
        }
    }
  }
}
