query GetOrganizationById($organizationId: ID!){
  organization(id: $organizationId) {
    id
    notes {
        content {
          id
          html
          content
          contentType
          createdAt
          updatedAt
          noted {
            __typename
            ... on Contact {
                id
                firstName
                lastName
            }
            __typename
            ... on Organization {
                id
                name
            }
          }
          source
          sourceOfTruth
          appSource
        }
    }
  }
}
