query GetCompanies($page: Int!, $limit: Int!){
  companies(pagination: {page: $page, limit: $limit}
            where:{filter: {property:"NAME" value:"openline" operation:CONTAINS}}) {
    totalPages
    totalElements
    content {
      id
      name
      addresses {
        id
        source
        country
        state
        city
        address
        address2
        zip
        phone
        fax
      }
    }
  }
}
