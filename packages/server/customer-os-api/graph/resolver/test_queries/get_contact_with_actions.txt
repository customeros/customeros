query GetContactById($contactId: ID!, $from: Time!, $to: Time!){
  contact(id: $contactId) {
    id
    actions(from:$from, to:$to) {
            __typename
            ... on PageViewAction {
              id
              application
              startedAt
              endedAt
              engagedTime
              pageUrl
              pageTitle
              orderInSession
              sessionId
            }
            __typename
            ... on MessageAction {
              id
              conversationId
              startedAt
              channel
            }
          }
  }
}
