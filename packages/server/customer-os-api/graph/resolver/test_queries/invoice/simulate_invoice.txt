mutation SimulateOnCycleInvoice($contractId: ID!, $serviceLines: [InvoiceSimulateServiceLineInput!]!) {
  invoice_Simulate(input: {
    contractId: $contractId
    serviceLines: $serviceLines
  }) {
    invoiceNumber
    postpaid
    offCycle
    invoicePeriodStart
    invoicePeriodEnd
    amount
    subtotal
    total
    invoiceLineItems {
      key
      description
      price
      quantity
      subtotal
      taxDue
      total
    }
  }
}
