query GetInvoice($page: Int!, $limit: Int!, $invoiceIssueDate: Any!) {
  invoices(pagination: {page: $page, limit: $limit}
              where: {AND:[
                            {filter: {property:"INVOICE_ISSUE_DATE" value: $invoiceIssueDate operation:EQ}}
                          ]
                     }
  )
  {
      totalPages
      totalElements
      content {
        metadata {
            id
        }
        invoiceNumber
        invoiceLineItems {
            description
        }
      }
  }
}
