query GetInvoice($page: Int!, $limit: Int!, $invoiceNumber: Any!) {
  invoices(pagination: {page: $page, limit: $limit}
              where: {AND:[
                            {filter: {property:"INVOICE_NUMBER" value: $invoiceNumber operation:EQ}}
                          ]
                     }
              sort:[{by: "NUMBER" direction:ASC }]
  )
  {
      totalPages
      totalElements
      content {
        metadata {
            id
        }
        invoiceNumber
        invoiceLineItems {
            description
        }
      }
  }
}
