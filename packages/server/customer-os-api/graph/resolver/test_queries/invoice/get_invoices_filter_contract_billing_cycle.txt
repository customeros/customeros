query GetInvoice($page: Int!, $limit: Int!, $contractBillingCycle: Any!) {
  invoices(pagination: {page: $page, limit: $limit}
              where: {AND:[
                            {filter: {property:"CONTRACT_BILLING_CYCLE" value: $contractBillingCycle operation:IN}}
                          ]
                     }
  )
  {
      totalPages
      totalElements
      content {
        metadata {
            id
        }
        invoiceNumber
        invoiceLineItems {
            description
        }
      }
  }
}
