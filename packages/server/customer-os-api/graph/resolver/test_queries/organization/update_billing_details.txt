mutation UpdateOrganization($organizationId: ID!) {
  organization_UpdateBillingDetails(
        input: {
            id: $organizationId
            amount: 100
            frequency: MONTHLY
            renewalCycle: ANNUALLY
            renewalCycleStart: "2020-01-01T11:12:35.000Z"
        }) {
      id
      name
      updatedAt
      sourceOfTruth
      accountDetails {
              billingDetails {
                  amount
                  frequency
                  renewalCycle
                  renewalCycleStart
                  renewalCycleNext
              }
              renewalLikelihood {
                  probability
                  previousProbability
                  comment
                  updatedAt
                  updatedById
                  updatedBy {
                        id
                        firstName
                        lastName
                  }
              }
              renewalForecast {
                  amount
                  potentialAmount
                  comment
                  updatedAt
                  updatedById
                  updatedBy {
                    id
                    firstName
                    lastName
                  }
              }
          }
    }
}
