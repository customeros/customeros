mutation UpdateOrganization($organizationId: ID!) {
  organization_UpdateRenewalForecast(
        input: {
            id: $organizationId
            amount: 100.01
            comment: "This is an updated comment"
        }) {
      id
      name
      updatedAt
      sourceOfTruth
      accountDetails {
              renewalForecast {
                  amount
                  potentialAmount
                  comment
                  updatedAt
                  updatedById
                  updatedBy {
                    id
                    firstName
                    lastName
                  }
              }
              renewalLikelihood {
                  probability
                  previousProbability
                  comment
                  updatedAt
                  updatedById
                  updatedBy {
                        id
                        firstName
                        lastName
                  }
              }
              billingDetails {
                  amount
                  frequency
                  renewalCycle
                  renewalCycleStart
              }
          }
    }
}
