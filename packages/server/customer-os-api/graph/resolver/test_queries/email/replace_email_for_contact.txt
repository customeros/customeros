mutation UpdateEmailForContact($contactId: ID!, $emailId: ID!) {
    emailUpdateInContact(contactId: $contactId,
                        input: {id: $emailId
                                label: HOME
                                primary: true
                                email: "new@email.com"
                        }) {
        id
        rawEmail
        validated
        email
        label
        primary
        createdAt
        updatedAt
    }
}
