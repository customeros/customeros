query GetMeeting($meetingId: ID!){
  meeting(id: $meetingId) {
       id
       name
       appSource
       start
       end
       source
       sourceOfTruth
       events {
         id
         createdAt
         eventIdentifier
         content
         contentType
         channel
         source
         sourceOfTruth
         appSource
         repliesTo {
           id
           eventIdentifier
           content
           contentType
           channel
         }
       }
  }
}
