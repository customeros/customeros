mutation CreateInteractionEvent($sessionId: ID, $replyTo: ID, $content: String, $contentType: String) {
interactionEvent_Create(
    event: {
        channel: "EMAIL"
        sentBy: [ {
              email: "sentBy@openline.ai"
        } ]
        sentTo: [ {
              email: "dest1@openline.ai"
        },
        {
              email: "dest2@openline.ai"
        }
        ]
        appSource: "Oasis"
        content: $content
        contentType: $contentType
        interactionSession: $sessionId
        repliesTo: $replyTo
    }
  ) {
    id
    createdAt
    content
    contentType
    channel
    interactionSession {
       id
       startedAt
       endedAt
       sessionIdentifier
       name
       status
       type
       channel
       source
       sourceOfTruth
       appSource
    }
    sentBy {
        __typename
        ... on EmailParticipant {
            emailParticipant {
                id
                rawEmail
            }
            type
        }
        ... on UserParticipant {
            userParticipant {
                id
                firstName
            }
            type
        }
        ... on PhoneNumberParticipant {
            phoneNumberParticipant {
                id
                rawPhoneNumber
            }
            type
        }
        ... on ContactParticipant {
            contactParticipant {
                id
                firstName
            }
            type
        }
    }
    sentTo {
        __typename
        ... on EmailParticipant {
            emailParticipant {
                id
                rawEmail
            }
            type
        }
        ... on UserParticipant {
            userParticipant {
                id
                firstName
            }
            type
        }
        ... on PhoneNumberParticipant {
            phoneNumberParticipant {
                id
                rawPhoneNumber
            }
            type
        }
        ... on ContactParticipant {
            contactParticipant {
                id
                firstName
            }
            type
        }
    }
    repliesTo {
      id
      eventIdentifier
      content
      contentType
      channel
    }
    source
    sourceOfTruth
    appSource
  }
}
