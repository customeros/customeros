query GetInteractionEvent($eventId: ID!){
  interactionEvent(id: $eventId) {
    id
    createdAt
    eventIdentifier
    content
    contentType
    channel
    actions {
        id
        content
        metadata
        actionType
    }
    interactionSession {
       id
       createdAt
       updatedAt
       identifier
       name
       status
       type
       channel
       source
       sourceOfTruth
       appSource
    }
    issue {
        id
        subject
    }
    repliesTo {
      id
      eventIdentifier
      content
      contentType
      channel
    }
    sentBy {
    __typename
        ... on EmailParticipant {
            emailParticipant {
            id
            rawEmail
            }
            type
        }
        ... on UserParticipant {
            userParticipant {
            id
            firstName
            }
            type
        }
        ... on PhoneNumberParticipant {
            phoneNumberParticipant {
            id
            rawPhoneNumber
            }
            type
        }
        ... on ContactParticipant {
            contactParticipant {
            id
            firstName
            }
            type
        }
    }
    sentTo {
    __typename
        ... on EmailParticipant {
            emailParticipant {
            id
            rawEmail
            }
            type
        }
        ... on UserParticipant {
            userParticipant {
            id
            firstName
            }
            type
        }
        ... on PhoneNumberParticipant {
            phoneNumberParticipant {
            id
            rawPhoneNumber
            }
            type
        }
        ... on ContactParticipant {
            contactParticipant {
            id
            firstName
            }
            type
        }
    }
    source
    sourceOfTruth
    appSource

  }
}
