mutation CreateInteractionEvent($meetingId: ID, $replyTo: ID, $content: String, $contentType: String, $eventType: String) {
interactionEvent_Create(
    event: {
        sentBy: [  ]
        sentTo: [ ]
        appSource: "Oasis"
        content: $content
        contentType: $contentType
        meetingId: $meetingId
        eventType: $eventType
        repliesTo: $replyTo
    }
  ) {
    id
    createdAt
    content
    contentType
    channel
    eventType
    interactionSession {
       id
       startedAt
       endedAt
       sessionIdentifier
       name
       status
       type
       channel
       source
       sourceOfTruth
       appSource
    }
    meeting {
        id
        name
    }
    sentBy {
        __typename
        ... on EmailParticipant {
            emailParticipant {
                id
                rawEmail
            }
            type
        }
        ... on UserParticipant {
            userParticipant {
                id
                firstName
            }
            type
        }
        ... on PhoneNumberParticipant {
            phoneNumberParticipant {
                id
                rawPhoneNumber
            }
            type
        }
        ... on ContactParticipant {
            contactParticipant {
                id
                firstName
            }
            type
        }
    }
    sentTo {
        __typename
        ... on EmailParticipant {
            emailParticipant {
                id
                rawEmail
            }
            type
        }
        ... on UserParticipant {
            userParticipant {
                id
                firstName
            }
            type
        }
        ... on PhoneNumberParticipant {
            phoneNumberParticipant {
                id
                rawPhoneNumber
            }
            type
        }
        ... on ContactParticipant {
            contactParticipant {
                id
                firstName
            }
            type
        }
    }
    repliesTo {
      id
      eventIdentifier
      content
      contentType
      channel
    }
    source
    sourceOfTruth
    appSource
  }
}
