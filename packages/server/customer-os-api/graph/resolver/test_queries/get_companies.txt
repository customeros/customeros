query GetCompanies($page: Int!, $limit: Int!){
  companies(pagination: {page: $page, limit: $limit}
            where:{filter: {property:"NAME" value:"oPeN" operation:CONTAINS}}
            sort:[{by: "NAME" direction:ASC }]) {
    totalPages
    totalElements
    content {
      id
      name
    }
  }
}
