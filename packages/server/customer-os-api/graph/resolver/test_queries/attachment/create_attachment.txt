mutation CreateAttachment($id: ID, $basePath: String!, $fileName: String!, $mimeType: String!, $size: Int64!) {
attachment_Create(
    input: {
        id: $id
        basePath: $basePath
        fileName: $fileName
        mimeType: $mimeType
        size: $size
        appSource: "Oasis"
    }
  ) {
      id
      createdAt
      basePath
      fileName
      mimeType
      size
      source
      sourceOfTruth
      appSource
    }
}
