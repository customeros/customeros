mutation CreateAnalysis($content: String, $contentType: String, $analysisType: String, $sessionId: ID, $eventId: ID) {
analysis_Create(
    analysis: {
        content: $content
        contentType: $contentType
        analysisType: $analysisType
        describes: {interactionSessionId: $sessionId, interactionEventId: $eventId}
        appSource: "Oasis"
    }
  ) {
      id
      createdAt

      content
      contentType
      analysisType
      describes {
        __typename
        ... on InteractionSession {
            id
            sessionIdentifier
            name
          }
        ... on InteractionEvent {
              id
              eventIdentifier
              contentType
              content
          }
      }
      source
      sourceOfTruth
      appSource
    }
}
