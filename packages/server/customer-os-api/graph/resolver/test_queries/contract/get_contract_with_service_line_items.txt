  contract(id: $contractId) {
    metadata {
        id
    }
    billingDetails {
        billingCycle
        invoicingStarted
        addressLine1
        addressLine2
        locality
        region
        country
        postalCode
        organizationLegalName
        billingEmail
        invoiceNote
        canPayWithCard
        canPayWithDirectDebit
        canPayWithBankTransfer
    }
    committedPeriods
    contractEnded
    contractName
    contractRenewalCycle
    contractSigned
    contractUrl
    currency
    billingEnabled
    serviceStarted
    contractStatus
    contractLineItems {
      id
      createdAt
      updatedAt
      name
      billed
      price
      quantity
      source
      sourceOfTruth
      appSource
      vatRate
    }
    createdBy {
        id
    }
  }
}
